        window.renderGrid = () => {
    const grid = document.getElementById('grid');
    const searchTerm = document.getElementById('search').value.toLowerCase();
    const category = document.getElementById('cat').value;

    let filtered = gameData.filter(g => {
        const matchesSearch = g.name.toLowerCase().includes(searchTerm);
        const matchesCat = (category === 'all') || (g.category === category) || (category === 'favs' && favs.includes(g.id));
        return matchesSearch && matchesCat;
    });

    // THIS IS THE NEW PART: Sort so Favs are always first
    filtered.sort((a, b) => {
        const aFav = favs.includes(a.id) ? 1 : 0;
        const bFav = favs.includes(b.id) ? 1 : 0;
        return bFav - aFav; // Higher "Fav" score moves to the front
    });

    grid.innerHTML = filtered.map(g => `
        <div class="card" onclick="window.playGame('${g.url}')">
            <div class="fav-area" onclick="event.stopPropagation(); window.toggleFav('${g.id}')">
                <span class="fav-btn ${favs.includes(g.id) ? 'active' : ''}">❤</span>
            </div>
            
            <div style="pointer-events: none; font-weight: bold; font-size: 13px; text-transform: uppercase; letter-spacing: 1.5px; line-height: 1.2;">
                ${g.name}
            </div>
        </div>
    `).join('');
};

        async function fetchUpdates() {
            const listEl = document.getElementById('updatesList');
            if (listEl) listEl.innerHTML = 'Loading updates...';
            onValue(ref(db, 'updates'), (s) => { renderUpdatesFromSnapshot(s); }, { onlyOnce: true });
        }

        function renderUpdatesFromSnapshot(snap) {
            const listEl = document.getElementById('updatesList');
            if (!listEl) return;
            if (!snap || !snap.exists()) { 
                listEl.innerHTML = '<div class="update-item">No updates yet.</div>'; 
                return; 
            }
            const items = [];
            snap.forEach(c => {
                const v = c.val() || {};
                items.push({ id: c.key, text: v.text, author: v.author, created: v.created });
            });
            items.sort((a,b) => (b.created||'').localeCompare(a.created||''));
            listEl.innerHTML = items.map(it => {
                const t = it.created ? new Date(it.created).toLocaleString() : '?';
                const auth = it.author || 'SYSTEM';
                return `<div class="update-item"><div>${escapeHtml(it.text||'')}</div><span class="update-time">${escapeHtml(auth)} • ${t}</span></div>`;
            }).join('');
        }
